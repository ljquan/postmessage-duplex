async function handleFetchData(request, client) {
  try {
    const response = await fetch(request.data.url)
    const data = await response.json()
    
    client.postMessage({
      requestId: request.requestId,
      ret: 0,
      data: data
    })
  } catch (error) {
    client.postMessage({
      requestId: request.requestId,
      ret: -1,
      msg: error.message
    })
  }
}
